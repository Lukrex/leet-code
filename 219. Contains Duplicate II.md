# My floating window O(n) solution
```py
class Solution:
    def containsNearbyDuplicate(self, nums: List[int], k: int) -> bool:
        numSet = set()
        for end in range(len(nums)):
            if end >= k+1:
                numSet.remove(nums[end-k-1])
            if nums[end] in numSet:
                return True
            numSet.add(nums[end])  
        return False
```

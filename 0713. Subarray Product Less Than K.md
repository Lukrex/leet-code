# An O(n) sliding-window solution after checking the Solutions tab
```py
class Solution:
    def numSubarrayProductLessThanK(self, nums: List[int], k: int) -> int:
        if k <= 1: return 0
        l = res = 0
        product = 1
        for r in range(len(nums)):
            product *= nums[r] # account for new right
            while product >= k: # make window valid from left
                product //= nums[l]
                l += 1
            res += 1 + (r-l) # add full subarray and it's subarrays to result
        return res
```

# My slow and veery complicated sliding window solution
```py
class Solution:
    def findLHS(self, nums: List[int]) -> int:
        nums.sort()
        numSet = set()
        longest = left = 0
        for right in range(len(nums)):
            if nums[right] not in numSet:
                if nums[right]-nums[left] > 1:
                    numSet.discard(nums[left])
                numSet.add(nums[right])
                while nums[left] not in numSet:
                    left += 1
            if len(numSet) == 2:
                a, b = tuple(numSet)
                if abs(a-b) == 1:
                    longest = max(longest, right-left+1)
        return longest
```

# Better AI generated sliding window solution
```py
class Solution:
    def findLHS(self, nums: List[int]) -> int:
        nums.sort()
        left = 0
        longest = 0

        for right in range(len(nums)):
            # shrink window until valid (max - min <= 1)
            while nums[right] - nums[left] > 1:
                left += 1

            # check if harmonic (max - min == 1)
            if nums[right] - nums[left] == 1:
                longest = max(longest, right - left + 1)

        return longest
```

# My O(n^2) solution
```py
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        if not s: return 0
        longest = 1
        try:
            for i in range(len(s)):
                str = ""
                current = i
                while s[current] not in str:
                   str += s[current] 
                   current += 1
                longest = max(len(str), longest)
        except:
            return max(len(str), longest) 
```

# A floating window O(n) solution from solutions tab
```py
class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        start = longest = 0
        charSet = set() 
        for end in range(len(s)):
            while s[end] in charSet:
                charSet.remove(s[start])
                start += 1
            charSet.add(s[end])
            longest = max(longest, len(charSet))
        return longest
```

# Quick O(n) sliding window solution
```py
class Solution:
    def minSubArrayLen(self, target: int, nums: List[int]) -> int:
        if sum(nums) < target: return 0
        minimum = len(nums)
        sums = left = 0
        for right in range(len(nums)):
            sums += nums[right]
            while sums >= target:
                minimum = min(minimum, right-left+1)
                sums -= nums[left]
                left += 1
        return minimum
```

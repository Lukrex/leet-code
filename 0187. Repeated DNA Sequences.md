# My AI assisted O(n) sliding window solution
```py
class Solution:
    def findRepeatedDnaSequences(self, s: str) -> List[str]:
        seen = set()
        res = set()
        left = 0
        for right in range(10, len(s)+1):
            window = s[left:right]
            if window in seen:
                res.add(window)
            else:
                seen.add(window)
            left += 1
        return list(res)
```

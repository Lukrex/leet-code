#My O(n log n) solution:
```py
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        if not nums: return 0
        nums = list(set(nums)) # set can not have duplicates, removes dupes
        nums.sort()
        numPrevious = -9999
        sequence = 1
        longestSequence = 1
        for numCurrent in nums:
            if numCurrent == numPrevious+1: sequence += 1
            else: sequence = 1
            if sequence > longestSequence: longestSequence = sequence
            numPrevious = numCurrent
        return longestSequence
```

#My assisted O(n) solution:
```py
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        if not nums: return 0
        nums = set(nums) # set can not have duplicates, removes dupes
        longest = 0
        for n in nums:
            if n-1 not in nums:
                length = 1
                current = n
                while current+1 in nums:
                    length += 1
                    current += 1
                longest = max(length, longest)
        return longest
```
